\section{Typographical Conventions}

\begin{itemize}
\item \textbf{Bold}: Vectors $\mathbf{v}$ or emphasis
\item \textit{Italic}: Scalars $x$, variables
\item $\mathcal{Script}$: Special spaces (e.g., $\mathcal{H}$ for Hilbert space)
\item $\mathbb{Blackboard}$: Number systems (e.g., $\mathbb{R}$, $\mathbb{C}$)
\item $\hat{Hat}$: Operators (e.g., $\hat{H}$ for Hamiltonian)
\item $\bar{Bar}$: Complex conjugate (e.g., $\bar{z}$) or anti-fields (e.g., $\bar{c}$)
\item $\tilde{Tilde}$: Fourier transform or dual object
\end{itemize}

\section{Index Conventions}

\begin{itemize}
\item Greek indices ($\mu, \nu, \lambda, \ldots$): Spacetime (0,1,2,3)
\item Latin indices ($i, j, k, \ldots$): Spatial (1,2,3)
\item Latin indices ($a, b, c, \ldots$): Gauge group (color, flavor)
\item Repeated indices: Einstein summation (e.g., $A^\mu B_\mu = \sum_{\mu=0}^3 A^\mu B_\mu$)
\end{itemize}

\section{Units and Conventions}

Throughout this book:
\begin{itemize}
\item Natural units: $\hbar = c = 1$
\item Metric signature: $(-, +, +, +)$ (mostly plus convention)
\item Planck mass: $M_{\text{Pl}} \approx 1.22 \times 10^{19}$ GeV
\item Temperatures in Kelvin unless specified
\item Energies in eV/GeV unless specified
\end{itemize}
